START:
    SET R0, 1               ; Set the R0 to 1
    SET R1, 1               ; Set the R1 to 1
    JMP LOOP                ; Jump to loop

LOOP:
    FWD R0                  ; Move forward by R0 value
    ADD R1, 1               ; Add 1 to R1
    JEQ STOP_MOVE, R1, 100  ; If R1 == 100, jump to STOP_MOVE else go through the loop again

STOP_MOVE:
    STOP           ; Stop the rover
    JMP END        ; Jump to END

END:
    ; End of program
